\section{Scenarios}\label{sec:scenarios}
\begin{comment}
	
		\noindent 

		\begin{itemize}
		    \item The user starts project planning by selecting Project--\textgreater Create Project Plan
		    \item The system presents a panel with text-boxes to enter project requirements
		    \item The user enters the collected requirements and scope definition
		    \item The user inputs the timeline for the project by selecting date.
		    \item In the budget section, the user enters the estimated cost of the project and budget resources.
  			\item Functional requirements are obtained from the user
  			\item The user adds the risk management section by clicking the add section button.
  			\item The user saves the project plan and exists the system
		\end{itemize}
        
	
\end{comment}
